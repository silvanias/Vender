389unsigned int loadTexture(char const *path)
390{
391 unsigned int textureID;
392 glGenTextures(1, &textureID);
393
394 int width;
395 int height;
396 int nrComponents;
397 unsigned char *data = stbi_load(path, &width, &height, &nrComponents, 0);
398 if (data)
399 {
400 GLenum format;
401 if (nrComponents == 1)
402 format = GL_RED;
403 else if (nrComponents == 3)
404 format = GL_RGB;
405 else if (nrComponents == 4)
406 format = GL_RGBA;
407
408 glBindTexture(GL_TEXTURE_2D, textureID);
409 glTexImage2D(GL_TEXTURE_2D, 0, format, width, height, 0, format, GL_UNSIGNED_BYTE, data);
410 glGenerateMipmap(GL_TEXTURE_2D);
411
412 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
413 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
414 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
415 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
416
417 stbi_image_free(data);
418 }
419 else
420 {
421 std::cout << "Texture failed to load at path: " << path << std::endl;
422 stbi_image_free(data);
423 }
424
425 return textureID;
426}
